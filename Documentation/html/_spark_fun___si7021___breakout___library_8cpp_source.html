129void Weather::changeResolution(uint8_t i)
130{
131 // Changes to resolution of ADDRESS measurements.
132 // Set i to:
133 // RH Temp
134 // 0: 12 bit 14 bit (default)
135 // 1: 8 bit 12 bit
136 // 2: 10 bit 13 bit
137 // 3: 11 bit 11 bit
138
139 uint8_t regVal = readReg();
140 // zero resolution bits
141 regVal &= 0b011111110;
142 switch (i) {
143 case 1:
144 regVal |= 0b00000001;
145 break;
146 case 2:
147 regVal |= 0b10000000;
148 break;
149 case 3:
150 regVal |= 0b10000001;
151 default:
152 regVal |= 0b00000000;
153 break;
154 }
155 // write new resolution settings to the register
156 writeReg(regVal);
157}

182uint16_t Weather::makeMeasurment(uint8_t command)
183{
184 // Take one ADDRESS measurement given by command.
185 // It can be either temperature or relative humidity
186 // TODO: implement checksum checking
187
188 uint16_t nBytes = 3;
189 // if we are only reading old temperature, read olny msb and lsb
190 if (command == 0xE0) nBytes = 2;
191
192 Wire.beginTransmission(ADDRESS);
193 Wire.write(command);
194 Wire.endTransmission();
195 // When not using clock stretching (*_NOHOLD commands) delay here
196 // is needed to wait for the measurement.
197 // According to datasheet the max. conversion time is ~22ms
198 delay(100);
199
200 Wire.requestFrom(ADDRESS,nBytes);
201 //Wait for data
202 int counter = 0;
203 while (Wire.available() < nBytes){
204 delay(1);
205 counter ++;
206 if (counter >100){
207 // Timeout: Sensor did not return any data
208 return 100;
209 }
210 }
211
212 unsigned int msb = Wire.read();
213 unsigned int lsb = Wire.read();
214 // Clear the last to bits of LSB to 00.
215 // According to datasheet LSB of RH is always xxxxxx10
216 lsb &= 0xFC;
217 unsigned int mesurment = msb << 8 | lsb;
218
219 return mesurment;
220}
