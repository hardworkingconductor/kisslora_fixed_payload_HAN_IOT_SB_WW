82// Internal function: Start watchdog timer
83// byte psVal - Prescale mask
84static void WDT_On (uint8_t psVal)
85{
86 // prepare timed sequence first
87 uint8_t ps = (psVal | (1<<WDIE)) & ~(1<<WDE);
88 cli();
89 wdt_reset();
90 // Clear WDRF in MCUSR
91 MCUSR &= ~(1<<WDRF);
92 // start timed sequence
93 WDTCSR |= (1<<WDCE) | (1<<WDE);
94 // set new watchdog timeout value
95 WDTCSR = ps;
96 sei();
97}

99// Internal function. Stop watchdog timer
100static void WDT_Off() {
101 cli();
102 wdt_reset();
103 // Clear WDRF in MCUSR
104 MCUSR &= ~(1<<WDRF);
105 // Write logical one to WDCE and WDE
106 // Keep old prescaler setting to prevent unintentional time-out
107 WDTCSR |= (1<<WDCE) | (1<<WDE);
108 // Turn off WDT
109 WDTCSR = 0x00;
110 sei();
111}

113// internal function.
114static int doSleep(long timeRem) {
115 uint8_t WDTps = 9; // WDT Prescaler value, 9 = 8192ms
116
117 isrcalled = 0;
118 sleep_enable();
119 while(timeRem > 0) {
120 //work out next prescale unit to use
121 while ((0x10<<WDTps) > timeRem && WDTps > 0) {
122 WDTps--;
123 }
124 // send prescaler mask to WDT_On
125 WDT_On((WDTps & 0x08 ? (1<<WDP3) : 0x00) | (WDTps & 0x07));
126 isrcalled=0;
127 while (isrcalled==0) {
128 // turn bod off
129 // MCUCR |= (1<<BODS) | (1<<BODSE);
130 //MCUCR &= ~(1<<BODSE); // must be done right before sleep
131 sleep_cpu(); // sleep here
132 }
133 // calculate remaining time
134 timeRem -= (0x10<<WDTps);
135 }
136 sleep_disable();
137 return timeRem;
138}
