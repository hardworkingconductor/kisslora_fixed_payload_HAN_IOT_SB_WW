151// \brief setup
152void setup(){
153 KISSLoRa_sleep_init();
154
155 // Initlialize serial
156 loraSerial.begin(57600);
157 debugSerial.begin(9600);
158
159 // Initialize LED outputs
160 pinMode(RGBLED_RED, OUTPUT);
161 pinMode(RGBLED_GREEN, OUTPUT);
162 pinMode(RGBLED_BLUE, OUTPUT);
163 pinMode(LED_LORA, OUTPUT);
164
165 // initialize rotary encoder
166 //Set pins as inputs
167 pinMode(ROTARY_PIN_0, INPUT);
168 pinMode(ROTARY_PIN_1, INPUT);
169 pinMode(ROTARY_PIN_2, INPUT);
170 pinMode(ROTARY_PIN_3, INPUT);
171 //Disable pullup resistors
172 digitalWrite(ROTARY_PIN_0, 0);
173 digitalWrite(ROTARY_PIN_1, 0);
174 digitalWrite(ROTARY_PIN_2, 0);
175 digitalWrite(ROTARY_PIN_3, 0);
176
177 // configure button
178 pinMode(BUTTON_PIN, INPUT); //Set pin as inputs
179 digitalWrite(BUTTON_PIN, 0); //Disable pullup resistors
180 //Attach an interrupt to the button pin - fire when button pressed down.
181 attachInterrupt(digitalPinToInterrupt(BUTTON_PIN), buttonPressedISR, FALLING);
182
183 //Initialize the I2C Si7021 sensor
184 sensor.begin();
185
186 // Wait a maximum of 10s for Serial Monitor
187 while (!debugSerial && millis() < 10000);
188
189 // Switch off leds
190 digitalWrite(RGBLED_RED, HIGH); //switch RGBLED_RED LED off
191 digitalWrite(RGBLED_GREEN, HIGH); //switch RGBLED_GREEN LED off
192 digitalWrite(RGBLED_BLUE, HIGH); //switch RGBLED_BLUE LED off
193 digitalWrite(LED_LORA, HIGH); //switch LED_LORA LED off
194
195 Wire.begin();
196 initAccelerometer();
197 setAccelerometerRange(ACC_RANGE);
198
199 // Initialize LoRaWAN radio
200 digitalWrite(RGBLED_RED, LOW); //switch RGBLED_RED LED on
201
202 ttn.onMessage(message); // Set callback for incoming messages
203 ttn.reset(true); // Reset LoRaWAN mac and enable ADR
204
205 debugSerial.println(F("-- STATUS"));
206 ttn.showStatus();
207
208 debugSerial.println(F("-- JOIN"));
209 ttn.join(appEui, appKey);
210
211 // initilize interval from rotary switch
212 nextInterval = getInitialInterval((uint8_t)getRotaryPosition());
213
214 digitalWrite(RGBLED_RED, HIGH); //switch RGBLED_RED LED off when join succeeds
215}

217// \brief mainloop
218void loop(){
219 debugSerial.println(F("-- LOOP"));
220
221 if(currentInterval != nextInterval){
222 debugSerial.print("Next interval set to: " + String(nextInterval/1000));
223 debugSerial.println(F(" Seconds"));
224 }
225 currentInterval = nextInterval;
226
227 digitalWrite(RGBLED_RED, HIGH); //switch RGBLED_RED LED off
228 digitalWrite(RGBLED_GREEN, HIGH); //switch RGBLED_GREEN LED off
229 digitalWrite(RGBLED_BLUE, HIGH); //switch RGBLED_BLUE LED off
230
231 // Measure Relative Humidity from the Si7021
232 float humidity = sensor.getRH();
233 debugSerial.print(F("Humidity: "));
234 debugSerial.print(humidity);
235 debugSerial.println(F(" %RH."));
236
237 // Measure Temperature from the Si7021
238 float temperature = sensor.getTemp();
239 // Temperature is measured every time RH is requested.
240 // It is faster, therefore, to read it from previous RH
241 // measurement with getTemp() instead with readTemp()
242 debugSerial.print(F("Temperature: "));
243 debugSerial.print(temperature);
244 debugSerial.println(F(" Degrees."));
245
246 // Measure luminosity
247 float luminosity = get_lux_value();
248 Serial.print(F("Ambient light: "));
249 Serial.print(luminosity);
250 Serial.println(F(" lux"));
251
252 // get rotary encode position
253 uint8_t rotaryPosition = (uint8_t)getRotaryPosition();
254 Serial.print(F("Rotary encoder position: "));
255 Serial.println(rotaryPosition);
256
258 getAcceleration(&x, &y, &z);
259 Serial.print(F("Acceleration:\tx="));
260 Serial.print(x);
261 Serial.print(F("g\n\t\ty="));
262 Serial.print(y);
263 Serial.print(F("g\n\t\tz="));
264 Serial.print(z);
265 Serial.println(F("g"));
266
268 float vdd = (float)ttn.getVDD()/1000;
269 Serial.print(F("RN2483 voltage: "));
270 Serial.print(vdd);
271 Serial.println(F(" Volt"));
272
273 // Wake RN2483
274 ttn.wake();
275
276 // Compose Cayenne message
277 lpp.reset(); // reset cayenne object
278
279 // add sensor values to cayenne data package
280 lpp.addTemperature(LPP_CH_TEMPERATURE, temperature);
281 lpp.addRelativeHumidity(LPP_CH_HUMIDITY, humidity);
282 lpp.addLuminosity(LPP_CH_LUMINOSITY, luminosity);
283 lpp.addDigital(LPP_CH_ROTARYSWITCH, rotaryPosition, IND);
284 lpp.addAccelerometer(LPP_CH_ACCELEROMETER, x, y, z);
285 lpp.addAnalog(LPP_CH_BOARDVCCVOLTAGE, vdd, INA);
286 lpp.addPresence(LPP_CH_PRESENCE, SAFE);
287 lpp.addAnalog(LPP_CH_SET_INTERVAL, (float)currentInterval/1000, OUTA);
288
289 digitalWrite(LED_LORA, LOW); //switch LED_LORA LED on
290
291 // send cayenne message on port 99
292 ttn.sendBytes(lpp.getBuffer(), lpp.getSize(), APPLICATION_PORT_CAYENNE);
293
294 // Set RN2483 to sleep mode
295 ttn.sleep(currentInterval - 100);
296 // This delay is not optional, try to remove it
297 // and say bye bye to your RN2483 sleep mode
298 delay(50);
299
300 digitalWrite(LED_LORA, HIGH); //switch LED_LORA LED off
301
302 // Set KISSLoRa to sleep.
303 sleep(currentInterval);
304}

348float get_lux_value(void){
349 int digital_value = analogRead(LIGHT_SENSOR_PIN);
350 double vlux = digital_value * (2.56/1023.0); //lux value in volts
351 double ilux = (vlux / 56) * 1000; //lux value in micro amperes
352 double lux = pow(10, (ilux / 10)); //Convert ilux to Lux value
353 return (float)lux; //Return Lux value as value without decimal
354}

437void getAcceleration(float *x, float *y, float *z){
438 // Resource: https://github.com/sparkfun/MMA8452_Accelerometer/blob/master/Libraries/Arduino/src/SparkFun_MMA8452Q.cpp
439 // Read the acceleration from registers 1 through 6 of the MMA8452 accelerometer.
440 // 2 registers per axis, 12 bits per axis.
441 // Bit-shifting right does sign extension to preserve negative numbers.
442 *x = ((short)(readAccelerometer(1)<<8 | readAccelerometer(2))) >> 4;
443 *y = ((short)(readAccelerometer(3)<<8 | readAccelerometer(4))) >> 4;
444 *z = ((short)(readAccelerometer(5)<<8 | readAccelerometer(6))) >> 4;
445
446 // Scale 12 bit signed values to units of g. The default measurement range is 2g.
447 // That is 11 bits for positive values and 11 bits for negative values.
448 // value = (value / (2^11)) * 2
449 *x = (float)*x / (float)(1<<11) * (float)(ACC_RANGE);
450 *y = (float)*y / (float)(1<<11) * (float)(ACC_RANGE);
451 *z = (float)*z / (float)(1<<11) * (float)(ACC_RANGE);
452}

457static void sleep(uint32_t delay_time_ms){
458 //Loop until delay is over, or if the push button is pressed, or if rotary switch is changed
459 while (delay_time_ms)
460 {
461 if(!USB_CABLE_CONNECTED){
462 KISSLoRa_sleep_delay_ms(100);
463 }else{
464 delay(100);
465 }
466
467 // Event: Send acknowledged message at alarm.
468 if(alarm){
469 debugSerial.println(F("-- ALARM!"));
470 digitalWrite(RGBLED_RED, LOW); //switch RGBLED_RED LED on
471
472 // Wake RN2483
473 ttn.wake();
474
475 lpp.reset();
476 lpp.addPresence(LPP_CH_PRESENCE, ALARM);
477 //lpp.addDigital(LPP_CH_SW_RELEASE, RELEASE, IN);
478 //lpp.addDigitalInput(LPP_CH_SW_RELEASE, RELEASE);
479 lpp.addDigital(LPP_CH_SW_RELEASE, RELEASE, IND);
480 //AAAAAAAAAAAAAAAAAAAAAAAAAAA
481
482
483 // Send it off
484 ttn.sendBytes(lpp.getBuffer(), lpp.getSize(), APPLICATION_PORT_CAYENNE, true);
485
486 // Set RN2483 to sleep mode
487 ttn.sleep(delay_time_ms - 2200);
488 // This delay is not optional, try to remove it
489 // and say bye bye to your RN2483 sleep mode
490 delay(50);
491
492 alarm = false;
493
494 digitalWrite(RGBLED_RED, HIGH); //switch RGBLED_RED LED off
495 }
496
497 if(delay_time_ms > 100){
498 delay_time_ms -= 100;
499 }else{
500 delay_time_ms = 0;
501 }
502 }
503}

510uint32_t getInitialInterval(uint8_t rotaryValue){
511 uint32_t intervalMs = REGULAR_INTERVAL;
512
513 rotaryValue &= INTERVAL_ROTARY_MASK;
514 switch(rotaryValue){
515 case INTERVAL_ROTARY_1:
516 intervalMs = INTERVAL_1;
517 break;
518
519 case INTERVAL_ROTARY_2:
520 intervalMs = INTERVAL_2;
521 break;
522
523 case INTERVAL_ROTARY_3:
524 intervalMs = INTERVAL_3;
525 break;
526
527 case INTERVAL_ROTARY_4:
528 intervalMs = INTERVAL_4;
529 break;
530
531 default:
532 intervalMs = REGULAR_INTERVAL;
533 break;
534 }
535 return intervalMs;
536}
